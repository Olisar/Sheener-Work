  `severity` tinyint(4) DEFAULT NULL CHECK (`severity` between 1 and 5),

  `ra_id` int(11) NOT NULL,
  PRIMARY KEY (`score_id`),
  KEY `aspect_id` (`aspect_id`),
  KEY `fk_env_impact_scores_ra` (`ra_id`),
  CONSTRAINT `env_impact_scores_ibfk_1` FOREIGN KEY (`aspect_id`) REFERENCES `env_aspects` (`aspect_id`) ON DELETE CASCADE,
  CONSTRAINT `fk_env_impact_scores_ra` FOREIGN KEY (`ra_id`) REFERENCES `env_ra` (`ra_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci

Table: risk_assessment
CREATE TABLE `risk_assessment` (
  `assessment_id` int(11) NOT NULL AUTO_INCREMENT,
  `risk_id` int(11) NOT NULL,
  `assessment_date` date NOT NULL,
  `assessed_by` int(11) NOT NULL COMMENT 'people_id FK',
  `inherent_likelihood` tinyint(1) NOT NULL COMMENT '1-5 scale',
  `inherent_severity` tinyint(1) NOT NULL COMMENT '1-5 scale',
  `inherent_risk_score` int(11) GENERATED ALWAYS AS (`inherent_likelihood` * `inherent_severity`) STORED COMMENT 'Auto-calculated',
  `inherent_risk_level` enum('Low','Medium','High','Critical') GENERATED ALWAYS AS (case when `inherent_likelihood` * `inherent_severity` <= 5 then 'Low' when `inherent_likelihood` * `inherent_severity` <= 10 then 'Medium' when `inherent_likelihood` * `inherent_severity` <= 20 then 'High' else 'Critical' end) STORED,
  `residual_likelihood` tinyint(1) DEFAULT NULL COMMENT 'After controls, 1-5 scale',
  `residual_severity` tinyint(1) DEFAULT NULL COMMENT 'After controls, 1-5 scale',
  `residual_risk_score` int(11) GENERATED ALWAYS AS (coalesce(`residual_likelihood`,0) * coalesce(`residual_severity`,0)) STORED COMMENT 'Auto-calculated',
  `residual_risk_level` enum('Low','Medium','High','Critical') GENERATED ALWAYS AS (case when coalesce(`residual_likelihood`,0) * coalesce(`residual_severity`,0) = 0 then 'Low' when coalesce(`residual_likelihood`,0) * coalesce(`residual_severity`,0) <= 5 then 'Low' when coalesce(`residual_likelihood`,0) * coalesce(`residual_severity`,0) <= 10 then 'Medium' when coalesce(`residual_likelihood`,0) * coalesce(`residual_severity`,0) <= 20 then 'High' else 'Critical' end) STORED,
  `risk_acceptability` enum('Acceptable','ALARP','Unacceptable') NOT NULL DEFAULT 'Unacceptable' COMMENT 'As Low As Reasonably Practicable',
  `assessment_methodology` enum('FMEA','HAZOP','Bowtie','FTA','HACCP','ICH Q9','Other') DEFAULT 'FMEA',
  `notes` text DEFAULT NULL,
  `approved_by` int(11) DEFAULT NULL COMMENT 'people_id FK',
  `approval_date` date DEFAULT NULL,
  `created_at` timestamp NOT NULL DEFAULT current_timestamp(),
  `updated_at` timestamp NOT NULL DEFAULT current_timestamp() ON UPDATE current_timestamp(),
  PRIMARY KEY (`assessment_id`),
  KEY `idx_risk_id` (`risk_id`),
  KEY `idx_assessed_by` (`assessed_by`),
  KEY `idx_assessment_date` (`assessment_date`),
  KEY `idx_inherent_risk_score` (`inherent_risk_score`),
  KEY `idx_residual_risk_score` (`residual_risk_score`),
  KEY `idx_approval_date` (`approval_date`),
  KEY `risk_assessment_ibfk_approved_by` (`approved_by`),
  CONSTRAINT `risk_assessment_ibfk_approved_by` FOREIGN KEY (`approved_by`) REFERENCES `people` (`people_id`) ON DELETE SET NULL,
  CONSTRAINT `risk_assessment_ibfk_assessed_by` FOREIGN KEY (`assessed_by`) REFERENCES `people` (`people_id`),
  CONSTRAINT `risk_assessment_ibfk_risk` FOREIGN KEY (`risk_id`) REFERENCES `risk_register` (`risk_id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=4 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci

  approved_by REFERENCES people(people_id)
